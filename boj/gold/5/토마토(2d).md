### 문제 이해
토마토 목록이 주어질때 모든 토마토가 익는 최소 날짜 구하기
입력이 MxN 2d array인게 차이점
### 추상화
bfs를 이용한 탐색 문제
### 계획
토마토 3d와 동일하나, 2차원 배열을 처리하도록 변경
### 코드
```python
import sys

input = sys.stdin.readline
from collections import deque


def solve(
    boxes: list[list[int]],
    start_pos: list[tuple[int, int]],
    total_cnt: int,
    M: int,
    N: int,
) -> int:
    ripe_cnt = len(start_pos)
    if total_cnt == ripe_cnt:
        return 0
    dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)]
    last_day = 0
    q = deque(start_pos)
    while q:
        x, y = q.popleft()
        current_day = boxes[y][x]
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < M and 0 <= ny < N:
                if boxes[ny][nx] == 0:
                    boxes[ny][nx] = current_day + 1
                    last_day = current_day + 1
                    ripe_cnt += 1
                    q.append((nx, ny))
    return -1 if total_cnt != ripe_cnt else last_day - 1


if __name__ == "__main__":
    [M, N] = map(int, input().split())
    boxes = []
    start_pos = []
    empty = 0
    for y in range(N):
        row = list(map(int, input().split()))
        for x in range(M):
            if row[x] == 1:
                start_pos.append((x, y))
            elif row[x] == -1:
                empty += 1
        boxes.append(row)
    total_cnt = M * N - empty
    print(solve(boxes, start_pos, total_cnt, M, N))

```
